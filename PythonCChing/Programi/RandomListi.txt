FUN generate_numbers(n)
    VAR numbers = []
    FOR i = 0 TO n THEN
        VAR num = RANDOM(1, 100)
        APPEND(numbers, num)
    END
    RETURN numbers
END



FUN printReverse(list)
    FOR i = LEN(list) - 1 TO -1 STEP -1 THEN
        PRINT(list / i + " ")
    END
    PRINT("\n")
END

FUN main()
    VAR n = INPUT_INT("Vnesi koliko Å¡tevil: ")
    VAR nums = generate_numbers(n)

    PRINT(LEN(nums))
    PRINT("\n ----------- \n")
    printReverse(nums)
END

main()
